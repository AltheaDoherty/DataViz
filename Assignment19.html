#8. Relationship between House Prices and Square Footage:

import pandas as pd
import matplotlib.pyplot as plt

# Load dataset
housing_data = pd.read_csv("housing_data.csv")

# Scatter Plot
plt.figure(figsize=(10, 6))
scatter = plt.scatter(housing_data['SquareFootage'], housing_data['Price'], c=housing_data['Price'], cmap="coolwarm", edgecolor="k", alpha=0.7)
plt.title('Scatter Plot of SquareFootage vs Price')
plt.xlabel('SquareFootage')
plt.ylabel('Price')
plt.colorbar(scatter, label='Size')
plt.show()

import seaborn as sns

# Load dataset

# Select only numerical columns for the heatmap
numerical_columns = housing_data.select_dtypes(include=['float64', 'int64']).columns

plt.figure(figsize=(10, 6))

#1. Proportion of Houses by Age Category:

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load dataset
housing_data = pd.read_csv("housing_data.csv")

# Pie Chart
housing_data['Age 1'] = 2023 - housing_data['YearBuilt']
housing_data['variable 3'] = pd.cut(housing_data['Age 1'], bins=[0, 10, 20, 30, 40, 50, 60, 70], 
                                     labels=['0-10 yrs', '11-20 yrs', '21-30 yrs', '31-40 yrs', '41-50 yrs', '51-60 yrs', '61-70 yrs'])
age_category_counts = housing_data['variable 3'].value_counts()
plt.figure(figsize=(8, 8))
age_category_counts.plot(kind='pie', autopct='%1.1f%%', startangle=140, colors=sns.color_palette('pastel'))
plt.title('Proportion of YearsBuilt')
plt.ylabel('')
plt.show()
