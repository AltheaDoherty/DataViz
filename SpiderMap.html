import geopandas as gpd
import pandas as pd
import matplotlib.pyplot as plt
from shapely.geometry import LineString

# Load city data
cities = pd.read_csv('fictionland_cities_with_real_coordinates.csv')

# Set the hub
capital_city_name = 'Boston'
capital_city = cities[cities['city_name'] == capital_city_name].iloc[0]

# Create a GeoDataFrame for cities
cities_gdf = gpd.GeoDataFrame(
    cities, geometry=gpd.points_from_xy(cities.longitude, cities.latitude), crs="EPSG:4326"
)

# Project cities to Web Mercator for easier mapping
cities_gdf = cities_gdf.to_crs(epsg=3857)

# Create LineStrings from the capital city to each other city
spider_connections = []
for idx, row in cities.iterrows():
    if row['city_name'] != capital_city_name:
        line = LineString([(capital_city['longitude'], capital_city['latitude']), 
                           (row['longitude'], row['latitude'])])
        spider_connections.append({'city_name': row['city_name'], 'geometry': line})

# Create a GeoDataFrame for the spider connections
spider_gdf = gpd.GeoDataFrame(spider_connections, geometry='geometry', crs="EPSG:4326").to_crs(epsg=3857)

# Plot the Spider Map
fig, ax = plt.subplots(figsize=(12, 8))

# Optional: load a U.S. base map for geographic context
us_states = gpd.read_file("https://raw.githubusercontent.com/PublicaMundi/MappingAPI/master/data/geojson/us-states.json").to_crs(epsg=3857)
us_states.plot(ax=ax, color="lightgrey", edgecolor="black")

# Plot all cities
cities_gdf.plot(ax=ax, color='blue', markersize=50, label="Cities")

# Highlight the capital city
capital_gdf = cities_gdf[cities_gdf['city_name'] == capital_city_name]
capital_gdf.plot(ax=ax, color='red', markersize=80, label="Capital City")

# Plot connections (lines) from the capital to other cities
spider_gdf.plot(ax=ax, color='green', linewidth=1.5, linestyle='--', alpha=0.8, label="Connections to Capital")

# Add title and legend
plt.title(f"Spider Map: Connections from {capital_city_name} to Other Cities")
plt.xlabel("Longitude")
plt.ylabel("Latitude")
plt.legend()
plt.show()
