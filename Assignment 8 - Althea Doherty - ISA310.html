#Highlighting Extreme Data Points in History Scores
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv("StudentPerformance-1.csv")
print(df.head())
fig, ax = plt.subplots(figsize=(6, 6))
bars = plt.bar(df["History_Score"], df["Semester"], color="pink")
bars[df["Semester"].idxmax()].set_color("skyblue")  # highest value in red
bars[df["Semester"].idxmin()].set_color("blue")  # lowest value in blue

plt.xlabel("History Score")
plt.ylabel("Semester")
plt.title("Student Performance")
plt.show()

#Comparing Math and English Scores Semester-wise
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv("StudentPerformance-1.csv")
print(df.head())
fig, ax = plt.subplots(figsize=(10, 6))

ax.barh(df["Semester"], -df["Math_Score"], color="skyblue", label="Math Scores")
ax.barh(df["Semester"], df["English_Score"], color="pink", label="English Scores")

plt.xlabel("Math Score")
plt.ylabel("Semester")
plt.title("Mirrored Horizontal Bar Chart of Math and English")
plt.show()
