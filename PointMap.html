import geopandas as gpd
import matplotlib.pyplot as plt
import pandas as pd
import contextily as ctx
import matplotlib.cm as cm
import numpy as np

# Load city data
cities = pd.read_csv('fictionland_cities_with_real_coordinates.csv')

# Convert DataFrame to GeoDataFrame with Point geometries
cities_gdf = gpd.GeoDataFrame(
    cities, geometry=gpd.points_from_xy(cities.longitude, cities.latitude), crs="EPSG:4326")

# Reproject to Web Mercator for compatibility with contextily basemaps
cities_gdf = cities_gdf.to_crs(epsg=3857)

# Generate a unique color for each city
num_cities = len(cities_gdf)
colors = cm.rainbow(np.linspace(0, 1, num_cities))  # Array of unique colors

# Plot city locations with individual colors
fig, ax = plt.subplots(figsize=(10, 10))
cities_gdf.plot(ax=ax, color='blue', markersize=40, label='City Locations')

# Add a basemap
ctx.add_basemap(ax, source=ctx.providers.OpenStreetMap.Mapnik)

# Add title and labels
plt.title("Point Map of Fictionland's Cities with Unique Colors")
plt.xlabel("Longitude")
plt.ylabel("Latitude")
plt.legend()
plt.show()
