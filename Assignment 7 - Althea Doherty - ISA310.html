import pandas as pd
import matplotlib.pyplot as plt

# Read data from the CSV file
data = pd.read_csv("StudentPerformance.csv")

# Sort values by Math_Score and create a stacked bar plot for selected subjects
data_sorted = data.sort_values("Math_Score", ascending=False)
data_sorted[["English_Score", "History_Score", "Science_Score"]].plot(kind="bar", stacked=True)

# Plot setup
plt.xlabel("Semesters sorted by Math Score")
plt.ylabel("Scores")
plt.title("Student Performance in English, History, and Science")
plt.xticks(rotation=90)

# Show the plot
plt.show()
